async function fetchWeather() { try { const response = await fetch('https://api.data.gov.sg/v1/environment/2-hour-weather-forecast'); const data = await response.json(); const forecasts = data.items[0].forecasts; const tableBody = document.getElementById('weather-data'); tableBody.innerHTML = ''; forecasts.forEach(forecast => { const row = `${forecast.area}${forecast.forecast}`; tableBody.innerHTML += row; }); } catch (error) { console.error("无法获取天气数据:", error); } } fetchWeather();
